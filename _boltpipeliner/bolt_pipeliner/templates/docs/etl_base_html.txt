    A --> B["load_data(input_table_names)"]
    B --> C["processed_df = process_data(input_tables)"]
    C --> U{{"unload?"}}
    U -- "No" --> Z["End"]
    U -- "Yes" --> U1["unload_data(processed_df)"]
    U1 --> U2["_ensure_namespace(save_catalog, fixed_schema)"]
    U2 --> U3["_apply_incremental_policy(processed_df)"]
    U3 --> U4["_create_table(df_to_write)"]
    U4 --> Z

      <table class="info-table" aria-describedby="etl-base usage tips">
        <tbody>

          <tr><th>Input contract</th><td><span class="kv">input_tables</span> is a dict keyed by YAML aliases, each value already loaded as a DataFrame.</td></tr>
          <tr><th>Incremental contract</th><td>For append/window modes, return data containing the configured <span class="kv">incremental_column</span> with valid <span class="kv">incremental_type</span>/<span class="kv">incremental_date_grain</span>.</td></tr>

    function setExpanded(expanded) {{
      codeBox.classList.toggle("collapsed", !expanded);
      toggleBtn.setAttribute("aria-expanded", String(expanded));
      toggleBtn.textContent = expanded ? "Hide full code" : "Show full code";
    }}

    toggleBtn.addEventListener("click", () => {{
      const expanded = toggleBtn.getAttribute("aria-expanded") === "true";
      setExpanded(!expanded);
    }});

    copyBtn.addEventListener("click", async () => {{
      const codeText = codeBox.querySelector("code").innerText;
      try {{
        await navigator.clipboard.writeText(codeText);
        copyBtn.textContent = "Copied!";
        setTimeout(() => (copyBtn.textContent = "Copy"), 1200);
      }} catch (e) {{
        copyBtn.textContent = "Press Ctrl/Cmd+C";
        setTimeout(() => (copyBtn.textContent = "Copy"), 1500);
      }}
    }});
